function afficherOffres(offresAff) { const container = document.getElementById("offres"); container.innerHTML = ""; if (offresAff.length === 0) { container.innerHTML = "

Aucune offre ne correspond à vos critères.

"; return; } offresAff.forEach(o => { const div = document.createElement("div"); div.className = "carte"; div.innerHTML = `

${o.title}

${o.cible ? `

Cible: ${o.cible}

` : ""} ${o.description ? `

${o.description}

` : ""} ${o.region ? `

Région: ${o.region}

` : ""} ${o.departement ? `

Département: ${o.departement}

` : ""} ${o.ville ? `

Ville: ${o.ville}

` : ""} ${o.deadline ? `

Deadline: ${o.deadline}

` : ""} ${o.tags ? `

Tags: ${o.tags.join(", ")}

` : ""} ${o.url ? `Voir plus` : ""} `; container.appendChild(div); }); }
